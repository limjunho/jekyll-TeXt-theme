### Code  
```cpp
#include <iostream>

int main()
{
    int N, X;
    int array[10000] = {0};

    scanf("%d %d", &N, &X);

    if (N < 1 || X > 10000)
    {
        return 0;
    }

    for (size_t i = 0; i < N; i++)
    {
        scanf("%d", &array[i]);
    }

    for (size_t i = 0; i < N; i++)
    {
        if (X > array[i])
        {
            printf("%d ", array[i]);
        }
    }

    return 0;
}
```
