### Code  
```cpp
#include <iostream>

int main()
{
    int T;
    int A, B;

    scanf("%d", &T);

    for (size_t i = 1; i <= T; i++)
    {
        scanf("%d %d", &A, &B);
        printf("Case #%d: %d\n", i, A + B);
    }

    return 0;
}
```
