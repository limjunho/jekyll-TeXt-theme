### Code  
```cpp
#include <cstdio>

int main()
{
    int N, max = 0;
    int score[1000] = {0};
    double average = 0;

    scanf("%d", &N);

    for (size_t i = 0; i < N; i++)
    {
        scanf("%d", &score[i]);
        if (max < score[i])
        {
            max = score[i];
        }
    }

    for (size_t i = 0; i < N; i++)
    {
        average += (double)score[i] / (double)max * 100;
    }
    average /= N;

    printf("%lf\n", average);

    return 0;
}
```
