### Code  
```cpp
#include <iostream>

int main()
{
    int N;

    scanf("%d", &N);

    if (N < 1 && N > 100)
    {
        return 0;
    }

    for (size_t i = 1; i <= N; i++)
    {
        for (size_t j = 0; j < i; j++)
        {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}
```
