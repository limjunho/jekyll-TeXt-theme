### Code  
```cpp
#include <cstdio>

int main()
{
    int ary[9] = {0};
    int max = -1, check = 0;

    for (size_t i = 0; i < 9; i++)
    {
        scanf("%d", &ary[i]);
        if (ary[i] > max)
        {
            max = ary[i];
            check = i + 1;
        }
    }

    printf("%d %d\n", max, check);

    return 0;
}
```
