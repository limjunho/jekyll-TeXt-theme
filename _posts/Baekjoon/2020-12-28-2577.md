### Code  
```cpp
#include <cstdio>

int main()
{
    int A, B, C, result, check;
    int count[10] = {0};

    scanf("%d %d %d", &A, &B, &C);
    result = A * B * C;

    while (result > 10)
    {
        check = result % 10;
        count[check]++;
        result = result / 10;
    }
    count[result]++;

    for (size_t i = 0; i < 10; i++)
    {
        printf("%d\n", count[i]);
    }

    return 0;
}
```
