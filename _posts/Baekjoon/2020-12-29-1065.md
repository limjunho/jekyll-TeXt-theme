### Code  
```cpp
#include <cstdio>

bool ishansu(int number)
{
    int first;
    int second;

    if (number >= 100)
    {
        first = number / 100 - number % 100 / 10;
        second = number % 100 / 10 - number % 10;

        if (first == second)
        {
            return true;
        }
        else
        {
            return false;
        }
    }
    else
    {
        return true;
    }
}

int main()
{
    int N, count = 0;

    scanf("%d", &N);

    for (size_t i = 1; i <= N; i++)
    {
        if (ishansu(i))
        {
            count++;
        }
    }

    printf("%d\n", count);

    return 0;
}
```
