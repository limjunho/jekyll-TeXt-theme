### Code  
```cpp
#include <cstdio>
#include <iostream>

int main()
{
    char S[100]={0};
    int alpha[26], index;

    // alpha배열을 모두 -1로 초기화
    std::fill_n(alpha, 26, -1);

    scanf("%s", S);

    for (size_t i = 0; i < 100; i++)
    {
        if (S[i] == '\0')
        {
            break;
        }
        index = (int)S[i] - 97;
        if (alpha[index] == -1)
        {
            alpha[index] = i;
        }
    }

    for (size_t i = 0; i < 26; i++)
    {
        printf("%d ", alpha[i]);
    }
    
    return 0;
}
```
