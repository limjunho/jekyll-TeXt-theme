#include <cstdio>

int main()
{
    char sentence[1000000] = {0};
    int count = 0;
    bool check = false;

    scanf("%[^\n]s", sentence);

    for (size_t i = 0; i < 1000000; i++)
    {
        if(sentence[i]=='\0'){
            if(sentence[i-1] == ' '){
                count--;
            }
            break;
        }
        if(sentence[i]==' '){
            count++;
        }
    }

    if(sentence[0] == ' '){
        printf("%d\n", count);
    }else{
        printf("%d\n", count+1);
    }

    return 0;
}
