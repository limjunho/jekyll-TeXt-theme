### Code  
```cpp
#include <cstdio>

int main()
{
    char number[100];
    int N, sum = 0;

    scanf("%d %s", &N, number);

    for (size_t i = 0; i < N; i++)
    {
        // 0의 ascii코드는 48이다.
        sum += (int)number[i] - 48;
    }

    printf("%d\n", sum);

    return 0;
}
```
