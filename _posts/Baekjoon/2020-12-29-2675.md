#include <cstdio>

int main()
{
    int T, R, index = 0;
    char S[20] = {0};

    scanf("%d", &T);

    for (size_t i = 0; i < T; i++)
    {
        scanf("%d %s", &R, S);

        for (size_t j = 0; j < 20; j++)
        {
            if (S[j] == '\0')
            {
                break;
            }
            for (size_t k = 0; k < R; k++)
            {
                printf("%c", S[j]);
            }
        }
        printf("\n");
    }

    return 0;
}
